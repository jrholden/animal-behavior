246    static function getPublicBooks( $only_non_catalog = false ) {

248

250        $in_catalog = Book::IN_CATALOG;
251
252        $sql_where_conditions = [ 'id != 1', 'public = 1' ];
253        if ( $only_non_catalog ) {

255        }
256

258        if ( ! empty( $sql_where_conditions ) ) {

264        }

266        $sql = "

268                b.blog_id AS id,
269                MAX(IF(b.meta_key='{$public}',CAST(b.meta_value AS UNSIGNED),null)) AS public,
270                MAX(IF(b.meta_key='{$in_catalog}',CAST(b.meta_value AS UNSIGNED),null)) AS inCatalog
271            FROM {$wpdb->blogmeta} b
272            GROUP BY id
273            {$sql_where}
