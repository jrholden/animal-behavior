741/**
742 * Render "Part" meta box
743 *
744 * @param |WP_Post $post
745 */
746function override_parent_id( $post ) {
747
748    if ( 'chapter' !== $post->post_type ) {
749        return; // Do nothing
750    }
751
752    if ( $post->post_parent ) {
753        $selected = $post->post_parent;
754    } elseif ( isset( $_GET['startparent'] ) ) {
755        $selected = absint( $_GET['startparent'] );
756    } else {
757        $selected = 0;
758    }
759
760    global $wpdb;
761    $results = $wpdb->get_results(
762        $wpdb->prepare(
763            "SELECT ID, post_title FROM {$wpdb->posts} WHERE post_type = 'part' AND post_status IN (%s, %s, %s, %s) ORDER BY menu_order ASC ",
764            [ 'draft', 'web-only', 'private', 'publish' ]
765        )
766    );
767
768    $aria_label = esc_attr( __( 'Part', 'pressbooks' ) );
769    $output = "<select name='parent_id' id='parent_id' aria-label='{$aria_label}'>\n";
770    foreach ( $results as $val ) {
771        $selected_html = ( (int) $selected === (int) $val->ID ) ? "selected='selected'" : '';
