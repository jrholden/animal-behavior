454    /**
455     * AsciiMath support.
456     *
457     * Backward compatibility support for "$asciimath $" shortcodes.
458     *
459     * $asciimath e^{i \pi} + 1 = 0$ -> [asciimath]e^{i \pi} + 1 = 0[/asciimath]
460     * $asciimath [a, b]$ -> [asciimath][a, b][/asciimath]
461     *
462     * @param $content
463     *
464     * @return string
465     */
466    public function dollarSignAsciiMathMarkup( $content ) {
467        $textarr = wp_html_split( $content );
468
469        $regex = '%
470            \$asciimath(?:=\s*|\s+)
471            ((?:
472                [^$]+ # Not a dollar
473            |
474                (?<=(?<!\\\\)\\\\)\$ # Dollar preceded by exactly one slash
475            )+)
476            (?<!\\\\)\$ # Dollar preceded by zero slashes
477        %ix';
478
479        foreach ( $textarr as &$element ) {
480            if ( '' === $element || '<' === $element[0] ) {
481                continue;
482            }
483
484            if ( false === stripos( $element, '$asciimath' ) ) {
485                continue;
486            }
487
488            $element = preg_replace_callback( $regex, [ $this, '_dollarSignAsciiMathSrc' ], $element );
489        }
490
491        return implode( '', $textarr );
492    }
493
494    /**
495     * Basically, a private method used by `preg_replace_callback` in `$this->dollarSignAsciiMathMarkup`
496     * (Can't be a real private method because `callable $callback`)
497     *
498     * @param $matches
499     *
500     * @return string
501     */
502    public function _dollarSignAsciiMathSrc( $matches ) {
503        $latex = $matches[1];
504        $latex = $this->latexEntityDecode( $latex );
505        // Remove unsupported fg, bg, size attributes
506        if ( preg_match( '/.+(&fg=[0-9a-f]{6}).*/i', $latex, $fg_matches ) ) {
507            $latex = str_replace( $fg_matches[1], '', $latex );
508        }
509        if ( preg_match( '/.+(&bg=[0-9a-f]{6}).*/i', $latex, $bg_matches ) ) {
510            $latex = str_replace( $bg_matches[1], '', $latex );
511        }
512        if ( preg_match( '/.+(&s=[0-9-]{1,2}).*/i', $latex, $s_matches ) ) {
513            $latex = str_replace( $s_matches[1], '', $latex );
514        }
515        return $this->asciiMathRender( $latex );
516    }
