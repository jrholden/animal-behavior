416    if ( $custom_styles && ! empty( $custom_styles->post_content ) ) {

462/**
463 * Add anchors to link insertion modal query results.

466 * @param array $query
467 *
468 * @return array
469 */
470function add_anchors_to_wp_link_query( $results, $query ) {
471    // Note to future-self: $results are paginated. If the user scrolls down, ajax is triggered, more function calls will happen.
472    $url = wp_parse_url( $_SERVER['HTTP_REFERER'] );
473    parse_str( $url['query'], $query );
474    $current_post_id = isset( $query['post'] ) ? $query['post'] : 0;
475    $new_results = [];
476    foreach ( $results as $result ) {
477        $new_results[] = $result;
478        $url = rtrim( $result['permalink'], '/' );
479        $post_id = $result['ID'];
480        $post = get_post( $post_id );
481        if ( $post && ! empty( trim( $post->post_content ) ) ) {
482            $html5 = new HtmlParser( true );
483            $doc = $html5->loadHTML( $post->post_content );
484            /** @var \DOMElement $node */
485            foreach ( $doc->getElementsByTagName( 'a' ) as $node ) {
486                if ( $node->hasAttribute( 'id' ) ) {
487                    $id_attribute = $node->getAttribute( 'id' );
488                    $permalink = ( (int) $current_post_id !== (int) $post->ID ) ? $url : '';
489                    $permalink .= "#{$id_attribute}";
490                    $new_results[] = [
491                        'ID' => $post->ID,
492                        'title' => '#' . $id_attribute . ' (' . $post->post_title . ')',
493                        'permalink' => $permalink,
494                        'info' => __( 'Internal Link', 'pressbooks' ),
495                    ];
496                }
497            }
498        }
499    }
500    return $new_results;
501}
