511        $src = '@' . $func( $fullpath );
512    }

528    $h_oversized = ( $orig_h > $max_h );

535    }

543        imagesavealpha( $dst, true );

546    imagedestroy( $src );

549    $func( $dst, $fullpath );
550    imagedestroy( $dst );
