292function put_in_storage( $data ) {
293    if ( function_exists( 'openssl_encrypt' ) ) {

297        $data = openssl_encrypt( $data, $method, NONCE_KEY, OPENSSL_RAW_DATA, $iv );
298        $data = $iv . $data;  // For storage/transmission, we concatenate the IV and cipher text

303/**
304 * @param string $data
305 *
306 * @return string
307 */
308function unpack_from_storage( $data ) {
309    // Check if there are valid base64 characters
310    if ( ! preg_match( '/^[a-zA-Z0-9\/\r\n+]*={0,2}$/', $data ) ) {
311        return false;
312    }
313    // Decode the string in strict mode and check the results
314    $decoded = base64_decode( $data, true );
315    if ( false === $decoded ) {
316        return false;
317    }
318    // Encode the string again
319    if ( base64_encode( $decoded ) !== $data ) {
320        return false;
321    }
322
323    if ( function_exists( 'openssl_decrypt' ) ) {
324        $method = 'aes-256-ctr';
325        $iv_size = openssl_cipher_iv_length( $method );
326        $iv = substr( $decoded, 0, $iv_size );
327        $data = @openssl_decrypt( substr( $decoded, $iv_size ), $method, NONCE_KEY, OPENSSL_RAW_DATA, $iv ); // @codingStandardsIgnoreLine
328    }
329    return $data;
